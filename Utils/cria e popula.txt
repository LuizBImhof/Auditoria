CREATE TABLE `CIDADE` ( `HANDLE` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `CEP` INTEGER NOT NULL UNIQUE, `NOME` TEXT NOT NULL, `ESTADO` TEXT NOT NULL );

CREATE TABLE "CLIENTE" ( `HANDLE` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `NOME` TEXT NOT NULL, `CIDADE` INTEGER NOT NULL, `BAIRRO` TEXT NOT NULL, `NUMERO_ENDERECO` INTEGER, `NASCIMENTO` TEXT NOT NULL, `SALDO` REAL, `ATIVO` INTEGER NOT NULL, 'CPF' INTEGER NOT NULL, FOREIGN KEY(`CIDADE`) REFERENCES `CIDADE`(`HANDLE`) );

CREATE TABLE `PRODUTO` ( `HANDLE` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `NOME` TEXT NOT NULL, `DESCRICAO` TEXT, `PRECO` REAL NOT NULL );

CREATE TABLE "USUARIO" ( `HANDLE` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `LOGIN` TEXT NOT NULL, `SENHA` TEXT NOT NULL, `EDITA` INTEGER NOT NULL, `APAGA` INTEGER NOT NULL, `INCLUE` INTEGER NOT NULL );

CREATE TABLE "VENDA" ( `HANDLE` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, `DATA` TEXT NOT NULL, `QUANTIDADE` INTEGER NOT NULL, `DATA_PGTO` INTEGER, `PRODUTO` INTEGER NOT NULL, `VALOR` INTEGER NOT NULL, `VENDEDOR` INTEGER NOT NULL, `CLIENTE` INTEGER NOT NULL, FOREIGN KEY(`CLIENTE`) REFERENCES `CLIENTE`(`HANDLE`), FOREIGN KEY(`PRODUTO`) REFERENCES `PRODUTO`(`HANDLE`), FOREIGN KEY(`VENDEDOR`) REFERENCES `USUARIO`(`HANDLE`) );
